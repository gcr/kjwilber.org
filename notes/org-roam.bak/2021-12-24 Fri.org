# Calculate K1 based on screen size: the maximum x-distance occurs
# roughly at the edge of the torus, which is at x=R1+R2, z=0.  we
# want that to be displaced 3/8ths of the width of the screen, which
# is 3/4th of the way from the center to the side of the screen.
# screen_width*3/8 = K1*(R1+R2)/(K2+0)
# screen_width*K2*3/(8*(R1+R2)) = K1
($screen_width*$K2*3/(8*($R1+$R2))) as $K1 |

def torus($A; $B):
    ($A|cos) as $cosA | ($B|cos) as $cosB |
    ($A|sin) as $sinA | ($B|sin) as $sinB |

    # theta goes around the cross-sectional circle of a torus
    range(0; 2*3.1415; 0.1) as $theta |
      ($theta|cos) as $costheta |
      ($theta|sin) as $sintheta |
      # phi goes around the center of revolution of a torus
      range(0; 2*3.1415; 0.05) as $phi |
          ($phi|cos) as $cosphi |
          ($phi|sin) as $sinphi |

          ($R2 + $R1*$costheta) as $circlex |
          ($R1 * $sintheta) as $circley |

          {x: ($circlex*($cosB*$cosphi + $sinA*$sinB*$sinphi) - $circley*$cosA*$sinB),
           y: ($circlex*($sinB*$cosphi - $sinA*$cosB*$sinphi) + $circley*$cosA*$cosB),
           z: ($K2 + $cosA*$circlex*$sinphi + $circley*$sinA),
           ooz: (1 / ($K2 + $cosA*$circlex*$sinphi + $circley*$sinA + 0.01)),
           L: ($cosphi*$costheta*$sinB - $cosA*$costheta*$sinphi -
           $sinA*$sintheta + $cosB*($cosA*$sintheta - $costheta*$sinA*$sinphi))
          }
    ;

def render:
    def _draw_point($dat; $pt):
      ($screen_width/2 + $K1*$pt.ooz*$pt.x|floor) as $xp |
      ($screen_height/2 - $K1*$pt.ooz*$pt.y|floor) as $yp |
      if ($pt.L <= 0 or
          $pt.ooz <= $dat.zbuffer[$yp][$xp]
      ) then $dat else
         {zbuffer: ($dat.zbuffer | .[$yp][$xp] = $pt.ooz),
          output: ($dat.output | .[$yp][$xp] = (".,-~:;=!*#$@"[$pt.L*8:$pt.L*8])
                  )
         }
      end
      ;
      reduce .[] as $point ({
         output: [range($screen_height) | [range($screen_width)|" "]],
         zbuffer: [range($screen_height) | [range($screen_width)|-9999]]
      }; _draw_point(.; $point)) |
      .output |.[] | join("")
    ;


foreach repeat(0) as $_ ({A: 1, B: 1}; {A: (.A+0.07), B: (.B+0.03)}; [torus(.A;.B)]|render)
